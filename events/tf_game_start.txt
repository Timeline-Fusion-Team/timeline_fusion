namespace = tf_game_start


# STK federal formation and contact for the good guys
event = {
	id = tf_game_start.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_playable_country = {
			limit = {
				has_origin = origin_STK_federation
			}
			add_tradition = tr_diplomacy_adopt
			add_tradition = tr_diplomacy_the_federation
			save_event_target_as = STK_federation_leader
		}
		every_playable_country = {
			limit = {
				is_good_federal_gang = yes
			}
			# Have met aliens (their liege)
			set_country_flag = first_contact_event
			set_country_flag = Story2 # Already encountered spacefarers

			every_playable_country = { # Ajout de tout les empires en communication et dans la fédération
				limit = {
					is_good_federal_gang = yes
					NOT = { event_target:STK_federation_leader }
				}
				establish_communications_no_message = event_target:STK_federation_leader
				join_alliance = {
					who = event_target:STK_federation_leader
					override_requirements = yes
				}
			} ##### Manque les opinions confiance et intels UNCOMPLETED
		}
		every_playable_country = {
			limit= {
				has_origin = origin_SGU_tauri
			}
			save_event_target_as = SGU_tauri_military_leader
			set_subject_of = {
				who = event_target:STK_federation_leader
			}
			random_agreement = {
			limit = {
				owner = { is_same_value = event_target:STK_federation_leader }
				target = { is_same_value = event_target:SGU_tauri_military_leader }
			}
			set_agreement_preset = { preset = preset_bulwark }
			set_agreement_terms = { subject_expand = subject_can_expand_with_tithe }
			}

			create_fleet = {
				effect = {
					set_owner = event_target:SGU_tauri_military_leader
					while = {
						count = 5
						create_ship = {
							name = random
							design = NAME_Ravager
						}
					}
					set_location = event_target:SGU_tauri_military_leader.capital_star
				}
			}

			give_technology = {
				tech = tech_destroyers
				message = yes
			}
			
		}
	}
}
